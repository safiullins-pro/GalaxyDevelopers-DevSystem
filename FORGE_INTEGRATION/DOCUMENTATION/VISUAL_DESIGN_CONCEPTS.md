    }
    50% { 
        box-shadow: 0 0 40px rgba(16, 185, 129, 0.6);
        transform: scale(1.02);
    }

    0% { background-position: -200% 0; }
    100% { background-position: 200% 0; }

    0% { opacity: 1; filter: grayscale(0%); }

@keyframes neural-spark {
    0%, 100% { opacity: 0.3; }
    50% { opacity: 1; }
}
